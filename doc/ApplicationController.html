def render_error(status, errors)
  @errors = errors
  status_code = status.is_a?(Fixnum) ? status : Rack::Utils::SYMBOL_TO_STATUS_CODE.fetch(status.to_sym, 500)
  render("errors/#{status_code}", status: status)
end
